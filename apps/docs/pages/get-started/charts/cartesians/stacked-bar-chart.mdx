## Anatomy

```tsx lines="2, 19-42"
import { Card } from "@wonderflow/react-components";
import { BarChart } from "@wonderflow/charts";
import { data } from "../api/.."

const data = [
         {
      continent: 'Europe',
      'QN95B Neo QLED 4K TV': 875,
      'Q800A Neo QLED 4K TV': 715,
      'Q60A QLED 4K TV': 184,
      'C2 OLED 4K TV': 428,
      'G2 OLED Evo 4K TV': 137,
      'B2 OLED 4K TV': 117,
    }, ...]

export default () => {
  return (
    <Card>
      <BarChart
        title="Products performance"
        subtitle="trends by Europe/America/Asia"
        data={data}
        sortBy="descending-value"
        isStacked={true}
        index={{
          dataKey: 'continent',
          label: 'Continent',
        }}
        series={{
          dataKey: [
            'QN95B Neo QLED 4K TV',
            'Q800A Neo QLED 4K TV',
            'Q60A QLED 4K TV',
            'C2 OLED 4K TV',
            'G2 OLED Evo 4K TV',
            'B2 OLED 4K TV',
          ],
          label: 'Product Units',
          domain: [0, 1000],
          hideZero: true,
        }}
        showAverage
      />
    </Card>
  );
};
```
