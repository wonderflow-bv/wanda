import { MdxLayout } from "@/components/layouts/mdx";
import { Stack, Text, Title, Grid } from "@wonderflow/react-components";
import { Tile } from "@/components/tile";
import { AssetResources } from "src/resources";

export default ({ children }) => (
  <MdxLayout
    title="Resources"
    description="Wonderflow related resources."
  >
    {children}
  </MdxLayout>
);

export const ResourceRender = () => {
  return (
    <Stack rowGap={80}>
      {AssetResources.map(section => (
        <Stack as="section" rowGap={24} key={section.title}>
          <Title as="h2" level="5">{section.title}</Title>
          <Grid
            columns={2}
            colMinWidth="20rem"
            columnGap={4}
            rowGap={4}
          >
            {section.items?.map(item => (
              <Grid.Item key={item.title}>
                <Tile
                  title={item.title}
                  description={item.description}
                  icon={item.icon}
                  as="a"
                  iconColor={item.iconColor}
                  href={item.url}
                  highlightOnHover
                  target={item.url.startsWith('http') ? '_blank' : undefined}
                  download={!item.url.startsWith('http')}>
                </Tile>
              </Grid.Item>
            ))}
          </Grid>
        </Stack>
      )
      )}
    </Stack>
  )
}

<ResourceRender />
